/auth
    POST /signup - Регистрация
        Вход: username, password
        Выход: user, token
    POST /signin
        Вход: username, password
        Выход: ( user, token, tfa: false ) или (user: {}, token: "", tfa: true) 
    POST /logout
        Вход: -
        Выход: message
    GET /user
        user, token
    POST /2fa
        Вход: username, password, code
        Выход: user, token
